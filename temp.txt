

//Create account table
CREATE TABLE `accounts` (
  `username` varchar(50) NOT NULL,
  `password` varchar(255) NOT NULL,
  `email` varchar(100) NOT NULL,
  `status` int DEFAULT '1',
  PRIMARY KEY (`username`),
  UNIQUE KEY `username` (`username`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

//Create accGroup table
CREATE TABLE `accgroups` (
  `groupName` varchar(255) NOT NULL,
  PRIMARY KEY (`groupName`),
  UNIQUE KEY `groupName` (`groupName`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci


//Create accountGroupRelation table
CREATE TABLE `accountsgroup` (
  `fk_username` varchar(100) NOT NULL,
  `fk_groupName` varchar(100) NOT NULL,
  KEY `fk_username` (`fk_username`),
  KEY `fk_groupName` (`fk_groupName`),
  CONSTRAINT `accountsgroup_ibfk_1` FOREIGN KEY (`fk_username`) REFERENCES `accounts` (`username`),
  CONSTRAINT `accountsgroup_ibfk_2` FOREIGN KEY (`fk_groupName`) REFERENCES `accgroups` (`groupName`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

//Create accountApplication table (KIV)
CREATE TABLE `accountApplication`(
   `fk_username` varchar(50) NOT NULL,
   `fk_App_Acronym` varchar(200) NOT NULL
   FOREIGN KEY (`fk_username`) REFERENCES `accounts` (`username`),
   FOREIGN KEY (`fk_App_Acronym`) REFERENCES `application` (App_Acronym)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

//Create application table
CREATE TABLE `application` (
  `App_Acronym` varchar(200) NOT NULL,
  `App_Description` mediumtext,
  `App_Rnumber` int DEFAULT NULL,
  `App_startDate` date DEFAULT NULL,
  `App_endDate` date DEFAULT NULL,
  `App_permit_Open` varchar(188) DEFAULT NULL,
  `App_permit_toDoList` varchar(188) DEFAULT NULL,
  `App_permit_Doing` varchar(188) DEFAULT NULL,
  `App_permit_Done` varchar(188) DEFAULT NULL,
  PRIMARY KEY (`App_Acronym`),
  UNIQUE KEY `App_Acronym` (`App_Acronym`),
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci


//Create plan table
CREATE TABLE `plan` (
  `Plan_MVP_name` varchar(200) NOT NULL,
  `Plan_startDate` date DEFAULT NULL,
  `Plan_endDAte` date DEFAULT NULL,
  `Plan_app_acronym` varchar(200) DEFAULT NULL,
  `colour` varchar(200) DEFAULT NULL,
  PRIMARY KEY (`Plan_MVP_name`),
  UNIQUE KEY `Plan_MVP_name` (`Plan_MVP_name`),
  KEY `Plan_app_acronym` (`Plan_app_acronym`),
  CONSTRAINT `plan_ibfk_1` FOREIGN KEY (`Plan_app_acronym`) REFERENCES `application` (`App_Acronym`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

//Create task table
CREATE TABLE `task` (
  `Task_name` varchar(200) NOT NULL,
  `Task_description` mediumtext,
  `Task_notes` long text,
  `Task_id` mediumtext,
  `Task_plan` varchar(200) DEFAULT NULL,
  `Task_app_Acronym` varchar(200) DEFAULT NULL,
  `Task_state` varchar(200) DEFAULT NULL,
  `Task_creator` varchar(200) DEFAULT NULL,
  `Task_owner` varchar(200) DEFAULT NULL,
  `Task_createDate` date DEFAULT NULL,
  PRIMARY KEY (`Task_name`),
  UNIQUE KEY `Task_name` (`Task_name`),
  KEY `Task_plan` (`Task_plan`),
  KEY `Task_app_Acronym` (`Task_app_Acronym`),
  CONSTRAINT `task_ibfk_1` FOREIGN KEY (`Task_plan`) REFERENCES `plan` (`Plan_MVP_name`),
  CONSTRAINT `task_ibfk_2` FOREIGN KEY (`Task_app_Acronym`) REFERENCES `application` (`App_Acronym`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
